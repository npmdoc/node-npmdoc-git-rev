function git-rev.log (cb)

  • description and source-code
    log = function (cb) {
  _command('git log --no-color --pretty=format:\'[ "%H", "%s", "%cr", "%an" ],\' --abbrev-commit', function (str) {
    str = str.substr(0, str.length-1)
    cb(JSON.parse('[' + str + ']'))
  })
}
